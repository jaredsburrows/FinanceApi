001package com.burrows.library.finance.api.model;
002
003import com.burrows.library.finance.api.model.yahoo.Property;
004import org.apache.commons.lang3.StringUtils;
005
006import java.util.Collection;
007import java.util.LinkedHashSet;
008import java.util.Set;
009
010/**
011 * Request URL Data for creating HTTP requests.
012 *
013 * @author <a href="mailto:jaredsburrows@gmail.com">Jared Burrows</a>
014 * @see Property
015 * @since 0.0.1
016 */
017public class Config {
018
019    private final static String YAHOO_FINANCE_BASE_URL = "http://download.finance.yahoo.com/d/quotes.csv?";
020    private final static String YAHOO_FINANCE_STOCKS = "s=";
021    private final static String YAHOO_FINANCE_PROPERTIES = "&f=";
022    private final static String YAHOO_FINANCE_OUTPUT_CSV = "&e=.csv";
023
024    private Set<String> properties = new LinkedHashSet<>();
025    private Set<String> quotes = new LinkedHashSet<>();
026    private Service service = Service.YAHOO;
027    private Output output = Output.CSV;
028
029    public enum Output {
030        CSV,
031        JSON,
032        XML
033    }
034
035    public enum Service {
036        GOOGLE,
037        YAHOO
038    }
039
040    public Output getOutput() {
041        return this.output;
042    }
043
044    public Config setOutput(final Output output) {
045        this.output = output;
046        return this;
047    }
048
049    public Service getService() {
050        return this.service;
051    }
052
053    public Config setService(final Service service) {
054        this.service = service;
055        return this;
056    }
057
058    public Set<String> getQuotes() {
059        return this.quotes;
060    }
061
062    public Config addQuote(final String quote) {
063        this.quotes.add(quote);
064        return this;
065    }
066
067    public Config addQuotes(final Collection<String> quotes) {
068        this.quotes.addAll(quotes);
069        return this;
070    }
071
072    public Set<String> getProperties() {
073        return this.properties;
074    }
075
076    public Config addProperty(final Property property) {
077        this.properties.add(property.getTagValue());
078        return this;
079    }
080
081    public Config addProperties(final Collection<Property> properties) {
082        for (final Property property : properties) {
083            this.properties.add(property.getTagValue());
084        }
085
086        return this;
087    }
088
089    public String getUrl() {
090        switch (this.service) {
091            default:
092            case YAHOO:
093                return YAHOO_FINANCE_BASE_URL + YAHOO_FINANCE_STOCKS + StringUtils.join(this.getQuotes(), ",")
094                        + YAHOO_FINANCE_PROPERTIES + StringUtils.join(this.getProperties(), "").replaceAll("0", "")
095                        + YAHOO_FINANCE_OUTPUT_CSV;
096        }
097    }
098}
