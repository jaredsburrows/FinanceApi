001package com.burrows.library.finance.api.core;
002
003import com.burrows.library.finance.api.model.Config;
004
005public interface YahooFormat extends Format {
006
007    class Impl implements YahooFormat {
008
009        final private Config config;
010
011        public Impl(final Config config) {
012            this.config = config;
013        }
014
015        @Override
016        public Quote outputCSV() {
017            return new Quote.Impl(this.config.setOutput(Config.Output.CSV));
018        }
019    }
020}
